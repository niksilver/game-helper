 67    def _set_unit_properties(self):
 68        """
 69        Set the various _mm and _px properties appropriately
 70        """
 71
 72        if self._width_px is None and self._width_mm is None:
 73            raise ValueError('Must specify width_mm or width_px')
 74        if not(self._width_px is None) and not(self._width_mm is None):
 75            raise ValueError('Cannot specify width of both mm and px')
 76        if self._unit == 'px' and not(self._width_px is None):
 77            raise ValueError('Cannot specify width_px when default unit is also px')
 78        if self._unit == 'mm' and not(self._width_mm is None):
 79            raise ValueError('Cannot specify width_mm when default unit is also mm')
 80
 81        match self._unit:
 82            case 'px':
 83                px_per_mm = self._width / self._width_mm
 84
 85                self._width_px  = int(self._width)
 86                self._width_mm  =     self._width_px / px_per_mm
 87                self._height_px = int(self._height)
 88                self._height_mm =     self._height_px / px_per_mm
 89                self._gutter_px = int(self._gutter)
 90                self._gutter_mm =     self._gutter_px / px_per_mm
 91            case 'mm':
 92                mm_per_px = self._width / self._width_px
 93
 94                self._width_mm  =     self._width
 95                self._width_px  = int(self._width_mm / mm_per_px)
 96                self._height_mm =     self._height
 97                self._height_px = int(self._height_mm / mm_per_px)
 98                self._gutter_mm =     self._gutter
 99                self._gutter_px = int(self._gutter_mm / mm_per_px)
100            case _:
101                raise ValueError(f"Cannot convert from unit '{self._unit}'")

348                   filename,
349                   size = None, width = None, height = None,
350                   ):
351        """
352        Load an image from a file, possibly resize it, and return it.
353        If `filename` points to an SVG it will be converted to a PNG.
354
355        Units must be in the default unit.
356        `im` may be an Image object or a filename. (If it's a filename
357        it may be an SVG.)
358        `size`, `width` and `height` are all optional.
359        `size` is a (width, height) pair.
360
361        The returned image will be RGBA.
362        """
363
364        # Make sure im is an Image in the correct format
365
366        im     = None
367        is_svg = False
368
369        if filename[-4:] == '.svg':
370            is_svg   = True
371            b_string = cairosvg.svg2png(url = filename)
372            b_io     = io.BytesIO(b_string)
373            im       = Image.open(b_io)
374            im       = im.convert('RGBA')
375
376        else:
377            im = Image.open(filename)
378            im = im.convert('RGBA')
379
380        # We may need to resize it.
381        # If it's an SVG we have to resize it as we (re)load it.
382        # Inconveniently, the SVG converter will only scale it in either the
383        # x or y dimension, so the resizing has to be done separately.
384
385        (resize, size_px) = self.need_resize_px(im, size, width, height)
386
387        if resize and is_svg:
388            max_size = max(size_px)
389            b_string = cairosvg.svg2png(url           = filename,
390                                        output_width  = max_size,
391                                        output_height = max_size,
392                                        )
393            b_io = io.BytesIO(b_string)
394            im   = Image.open(b_io)
395            im   = im.convert('RGBA')
396            im   = im.resize(size = size_px)
397
398        elif resize:
399            im = im.resize(size = size)
400
401        return im

404    def paste(self,
405              im,
406              size = None, width = None, height = None,
407              x_left = None, x_centre = None, x_right = None,
408              y_top = None, y_middle = None, y_bottom = None):
409        """
410        Paste a given image onto the card; it will use itself as a mask.
411        (0, 0) is the top left of the card, excluding the gutters.
412
413        See `load_image` for explanation of
414        `size`, `width` and `height`.
415        """
416
417        # Make sure we have an Image and it's the specified size.
418
419        if type(im) == str:
420            im = self.load_image(im, size, width, height)
421
422        else:
423            (resize, size_px) = self.need_resize_px(im, size, width, height)
424            if resize:
425                im = im.resize(size = size_px)
426
427        # Switch to pixels
428
429        x_left   = self.to_px(x_left)
430        x_centre = self.to_px(x_centre)
431        x_right  = self.to_px(x_right)
432        y_top    = self.to_px(y_top)
433        y_middle = self.to_px(y_middle)
434        y_bottom = self.to_px(y_bottom)
435
436        x_pos, y_pos = None, None
437
438        if not(x_left is None):
439            x_pos = x_left + self._gutter_px
440        if not(x_right is None):
441            x_pos = x_right - im.width + self._gutter_px
442        if not(x_centre is None):

459    def need_resize_px(self, im, size = None, width = None, height = None):
460        """
461        Decide if an image should be resized, and what its new dimensions
462        should be, in pixels.
463        Only one of `size`, `width` and `height` is needed, and the dimensions
464        are the default dimensions of the object.
465        If only `width` or `height` is specified the resizing will be a
466        proportional scaling.
467
468        Returns (flag, (width, height)) where flag is if it requires resizing.
469        """
470        # Convert to pixels
471
472        if size is not None:
473            size = (self.to_px(size[0]),
474                    self.to_px(size[1]) )
475        if width is not None:
476            width = self.to_px(width)
477        if height is not None:
478            height = self.to_px(height)
479
480        # Switch the values into width and height only (either may be None)
481
482        if size is not None:
483            (width, height) = size
484
485        # Should we resize?
486
487        resize = False
488        if (width is not None) and im.width != width:
489            resize = True
490        if (height is not None) and im.height != height:
491            resize = True
492
493        # Calculate desired width and height
494
495        if (width is None) and (height is None):
496            return (resize, im.size)
497
498        if (width is not None) and (height is None):
499            height = im.height * (width / im.width)
500
501        if (width is None) and (height is not None):
502            width = im.width * (height / im.height)
503
504        return (resize, (int(width), int(height)))

640    def _insert_new_lines(text, len):
641        """
642        Given a text string and a line length, replace spaces with new line
643        characters to fit the length.
644        """
645        space = 0    # Index of last space
646        count = 0    # Number of characters consumed
647
648        for i, char in enumerate(text):
649            count = count + 1
650            if char == ' ':
651                space = i
652            elif char == '\n':
653                text = text[:i] + '\n' + text[i+1:]
654                count = 0
655            if count+1 == len:
656                text = text[:space] + '\n' + text[space+1:]
657                count = i - space
658
659        return text

348    def load_image(self,
349                   filename,
350                   size = None, width = None, height = None,
351                   ):
352        """
353        Load an image from a file, possibly resize it, and return it.
354        If `filename` points to an SVG it will be converted to a PNG.
355
356        Units must be in the default unit.
357        `im` may be an Image object or a filename. (If it's a filename
358        it may be an SVG.)
359        `size`, `width` and `height` are all optional.
360        `size` is a (width, height) pair.
361
362        The returned image will be RGBA.
363        """
364
365        # Make sure im is an Image in the correct format
366
367        im     = None
368        is_svg = False
369
370        if filename[-4:] == '.svg':
371            is_svg   = True
372            b_string = cairosvg.svg2png(url = filename)
373            b_io     = io.BytesIO(b_string)
374            im       = Image.open(b_io)
375            im       = im.convert('RGBA')
376
377        else:
378            im = Image.open(filename)
379            im = im.convert('RGBA')
380
381        # We may need to resize it.
382        # If it's an SVG we have to resize it as we (re)load it.
383        # Inconveniently, the SVG converter will only scale it in either the
384        # x or y dimension, so the resizing has to be done separately.
385
386        (resize, size_px) = self.need_resize_px(im, size, width, height)
387
388        if resize and is_svg:
389            max_size = max(size_px)
390            b_string = cairosvg.svg2png(url           = filename,
391                                        output_width  = max_size,
392                                        output_height = max_size,
393                                        )
394            b_io = io.BytesIO(b_string)
395            im   = Image.open(b_io)
396            im   = im.convert('RGBA')
397            im   = im.resize(size = size_px)
398
399        elif resize:
400            im = im.resize(size = size)
401
402        return im

460    def need_resize_px(self, im, size = None, width = None, height = None):
461        """
462        Decide if an image should be resized, and what its new dimensions
463        should be, in pixels.
464        Only one of `size`, `width` and `height` is needed, and the dimensions
465        are the default dimensions of the object.
466        If only `width` or `height` is specified the resizing will be a
467        proportional scaling.
468
469        Returns (flag, (width, height)) where flag is if it requires resizing.
470        """
471        # Convert to pixels
472
473        if size is not None:
474            size = (self.to_px(size[0]),
475                    self.to_px(size[1]) )
476        if width is not None:
477            width = self.to_px(width)
478        if height is not None:
479            height = self.to_px(height)
480
481        # Switch the values into width and height only (either may be None)
482
483        if size is not None:
484            (width, height) = size
485
486        # Should we resize?
487
488        resize = False
489        if (width is not None) and im.width != width:
490            resize = True
491        if (height is not None) and im.height != height:
492            resize = True
493
494        # Calculate desired width and height
495
496        if (width is None) and (height is None):
497            return (resize, im.size)
498
499        if (width is not None) and (height is None):
500            height = im.height * (width / im.width)
501
502        if (width is None) and (height is not None):
503            width = im.width * (height / im.height)
504
505        return (resize, (int(width), int(height)))

642        """
643        Given a text string and a line length, replace spaces with new line
644        characters to fit the length.
645        """
646        space = 0    # Index of last space
647        count = 0    # Number of characters consumed
648

651            if char == ' ':
652                space = i
653            elif char == '\n':
654                text = text[:i] + '\n' + text[i+1:]
655                count = 0
656            if count+1 == len:
657                text = text[:space] + '\n' + text[space+1:]
658                count = i - space

475                    self.to_px(size[1]) )

482

484            (width, height) = size
485
486        # Should we resize?
487

489        if (width is not None) and im.width != width:

501

503            width = im.width * (height / im.height)
504
